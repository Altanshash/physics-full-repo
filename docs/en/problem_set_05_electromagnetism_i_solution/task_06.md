# Problem 6 – 2D Field Map

## Given

System: any three charges in a plane.

Find:

1. Implement a function calculating the electric field $\vec{E}(x,y)$.
2. Generate a vector field map.
3. Find the equilibrium point numerically.
4. Investigate the stability of the equilibrium point.
5. Compare with the case of two charges.

Requirement: HTML visualization with the ability to change the positions of the charges.

---

## Solution

For one point charge, the electric field is:

```math
\vec{E}
=
kq\frac{\vec{r}}{|\vec{r}|^3}
```

For three charges:

```math
\vec{E}(x,y)
=
\sum_{i=1}^{3}
kq_i
\frac{(x-x_i,\ y-y_i)}
{\left((x-x_i)^2+(y-y_i)^2\right)^{3/2}}
```

Where:

```math
\vec{r}_i=(x-x_i,\ y-y_i)
```

and

```math
r_i=\sqrt{(x-x_i)^2+(y-y_i)^2}
```

---

## 1. Function for electric field

The field components are:

```math
E_x(x,y)
=
\sum_{i=1}^{3}
kq_i
\frac{x-x_i}
{\left((x-x_i)^2+(y-y_i)^2\right)^{3/2}}
```

```math
E_y(x,y)
=
\sum_{i=1}^{3}
kq_i
\frac{y-y_i}
{\left((x-x_i)^2+(y-y_i)^2\right)^{3/2}}
```

---

## 2. Vector field map

A vector field map is generated by calculating $\vec{E}(x,y)$ at many grid points and drawing arrows.

---

## 3. Equilibrium point

An equilibrium point satisfies:

```math
\vec{E}(x,y)=0
```

or:

```math
E_x(x,y)=0
```

```math
E_y(x,y)=0
```

Since it is difficult to solve analytically for three charges, the equilibrium point is found numerically by searching the point where:

```math
|\vec{E}|=\sqrt{E_x^2+E_y^2}
```

is minimum.

---

## 4. Stability

To check stability, move a small test charge slightly away from the equilibrium point.

- If the force pulls it back, the equilibrium is stable.
- If the force pushes it farther away, the equilibrium is unstable.

For electrostatic systems of fixed point charges, equilibrium points are usually unstable.

---

## 5. Comparison with two charges

For two charges, the field pattern is simpler and often symmetric.

For three charges:

- the field is more complex;
- equilibrium points can shift;
- the field map depends strongly on charge positions and signs.

<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

const k = 9e9;
const scale = 80;
const originX = canvas.width / 2;
const originY = canvas.height / 2;

let charges = [];

function toPixelX(x) {
  return originX + x * scale;
}

function toPixelY(y) {
  return originY - y * scale;
}

function toWorldX(px) {
  return (px - originX) / scale;
}

function toWorldY(py) {
  return -(py - originY) / scale;
}

function readCharges() {
  charges = [
    {
      q: Number(document.getElementById("q1").value) * 1e-6,
      x: Number(document.getElementById("x1").value),
      y: Number(document.getElementById("y1").value)
    },
    {
      q: Number(document.getElementById("q2").value) * 1e-6,
      x: Number(document.getElementById("x2").value),
      y: Number(document.getElementById("y2").value)
    },
    {
      q: Number(document.getElementById("q3").value) * 1e-6,
      x: Number(document.getElementById("x3").value),
      y: Number(document.getElementById("y3").value)
    }
  ];
}

function electricField(x, y) {
  let Ex = 0;
  let Ey = 0;

  for (let c of charges) {
    const dx = x - c.x;
    const dy = y - c.y;
    const r2 = dx * dx + dy * dy;

    if (r2 < 0.01) continue;

    const r = Math.sqrt(r2);
    const r3 = r2 * r;

    Ex += k * c.q * dx / r3;
    Ey += k * c.q * dy / r3;
  }

  return { Ex, Ey };
}

function fieldMagnitude(x, y) {
  const E = electricField(x, y);
  return Math.sqrt(E.Ex * E.Ex + E.Ey * E.Ey);
}

function findEquilibrium() {
  let bestX = 0;
  let bestY = 0;
  let bestMag = Infinity;

  for (let x = -4; x <= 4; x += 0.05) {
    for (let y = -3; y <= 3; y += 0.05) {
      let tooClose = false;

      for (let c of charges) {
        const dx = x - c.x;
        const dy = y - c.y;
        if (Math.sqrt(dx * dx + dy * dy) < 0.25) {
          tooClose = true;
        }
      }

      if (tooClose) continue;

      const mag = fieldMagnitude(x, y);

      if (mag < bestMag) {
        bestMag = mag;
        bestX = x;
        bestY = y;
      }
    }
  }

  return { x: bestX, y: bestY, mag: bestMag };
}

function checkStability(eq) {
  const h = 0.05;

  const E1 = electricField(eq.x + h, eq.y);
  const E2 = electricField(eq.x - h, eq.y);
  const E3 = electricField(eq.x, eq.y + h);
  const E4 = electricField(eq.x, eq.y - h);

  const dot1 = E1.Ex * h;
  const dot2 = E2.Ex * (-h);
  const dot3 = E3.Ey * h;
  const dot4 = E4.Ey * (-h);

  const total = dot1 + dot2 + dot3 + dot4;

  if (total < 0) {
    return "approximately stable";
  } else {
    return "unstable";
  }
}

function drawArrow(x1, y1, x2, y2, color) {
  const angle = Math.atan2(y2 - y1, x2 - x1);

  ctx.beginPath();
  ctx.moveTo(x1, y1);
  ctx.lineTo(x2, y2);
  ctx.strokeStyle = color;
  ctx.lineWidth = 2;
  ctx.stroke();

  ctx.beginPath();
  ctx.moveTo(x2, y2);
  ctx.lineTo(
    x2 - 7 * Math.cos(angle - Math.PI / 6),
    y2 - 7 * Math.sin(angle - Math.PI / 6)
  );
  ctx.lineTo(
    x2 - 7 * Math.cos(angle + Math.PI / 6),
    y2 - 7 * Math.sin(angle + Math.PI / 6)
  );
  ctx.closePath();
  ctx.fillStyle = color;
  ctx.fill();
}

function drawAxes() {
  ctx.strokeStyle = "#ccc";
  ctx.lineWidth = 1;

  ctx.beginPath();
  ctx.moveTo(0, originY);
  ctx.lineTo(canvas.width, originY);
  ctx.moveTo(originX, 0);
  ctx.lineTo(originX, canvas.height);
  ctx.stroke();

  ctx.fillStyle = "#555";
  ctx.font = "13px Arial";
  ctx.fillText("x", canvas.width - 20, originY - 8);
  ctx.fillText("y", originX + 8, 20);
}

function drawFieldMap() {
  const step = 45;

  for (let px = 35; px < canvas.width; px += step) {
    for (let py = 35; py < canvas.height; py += step) {
      const x = toWorldX(px);
      const y = toWorldY(py);

      const E = electricField(x, y);
      const mag = Math.sqrt(E.Ex * E.Ex + E.Ey * E.Ey);

      if (mag === 0) continue;

      const nx = E.Ex / mag;
      const ny = E.Ey / mag;

      const length = Math.min(22, 6 + Math.log10(mag + 1) * 4);

      drawArrow(
        px,
        py,
        px + nx * length,
        py - ny * length,
        "#2c7be5"
      );
    }
  }
}

function drawCharges() {
  for (let c of charges) {
    const px = toPixelX(c.x);
    const py = toPixelY(c.y);

    ctx.beginPath();
    ctx.arc(px, py, 18, 0, Math.PI * 2);

    if (c.q > 0) {
      ctx.fillStyle = "#e74c3c";
    } else {
      ctx.fillStyle = "#3498db";
    }

    ctx.fill();
    ctx.strokeStyle = "#222";
    ctx.stroke();

    ctx.fillStyle = "white";
    ctx.font = "bold 18px Arial";
    ctx.textAlign = "center";
    ctx.textBaseline = "middle";

    if (c.q > 0) {
      ctx.fillText("+", px, py);
    } else {
      ctx.fillText("-", px, py);
    }

    ctx.textAlign = "left";
    ctx.textBaseline = "alphabetic";
  }
}

function drawEquilibrium(eq) {
  const px = toPixelX(eq.x);
  const py = toPixelY(eq.y);

  ctx.beginPath();
  ctx.arc(px, py, 7, 0, Math.PI * 2);
  ctx.fillStyle = "#111";
  ctx.fill();

  ctx.font = "14px Arial";
  ctx.fillStyle = "#111";
  ctx.fillText("Equilibrium", px + 10, py - 10);
}

function update() {
  readCharges();

  ctx.clearRect(0, 0, canvas.width, canvas.height);

  drawAxes();
  drawFieldMap();
  drawCharges();

  const eq = findEquilibrium();
  const stability = checkStability(eq);

  drawEquilibrium(eq);

  document.getElementById("info").innerHTML =
    "Equilibrium point: x = " + eq.x.toFixed(2) +
    " m, y = " + eq.y.toFixed(2) +
    " m | |E| ≈ " + eq.mag.toExponential(2) +
    " N/C | Stability: " + stability;
}

function resetValues() {
  document.getElementById("q1").value = 3;
  document.getElementById("x1").value = -2;
  document.getElementById("y1").value = 0;

  document.getElementById("q2").value = -2;
  document.getElementById("x2").value = 2;
  document.getElementById("y2").value = 0;

  document.getElementById("q3").value = 2;
  document.getElementById("x3").value = 0;
  document.getElementById("y3").value = 2;

  update();
}

update();
</script>

</html>

## Conclusion

The electric field map is obtained by adding the fields from all three charges.  
The equilibrium point is the point where the total electric field is approximately zero.  
For three charges, the field pattern is more complex than for two charges, and the equilibrium point depends on the charge values and positions.
